    pop dword ptr [eax]
    mov [eax+4], esp
    mov [eax+8], ebx
    mov [eax+12], ebp

    mov ecx, [esp+4]
    mov ebp, [ecx+12]
    mov ebx, [ecx+8]
    mov esp, [ecx+4]

    push [ecx]
    ret
}

}
